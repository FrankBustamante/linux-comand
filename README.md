# linux-commands

- tar.gz  
  tar -czvf empaquetado.tar.gz /carpeta/a/empaquetar/   
  tar -xzvf archivo.tar.gz   
 
- tar  
  tar -cvf paquete.tar /dir/a/comprimir/

- gz  
  gzip -9 archivo  
  gzip -d archivo.gz    
  
- zip  
  zip archivo.zip carpeta   
  unzip archivo.zip
